CREATE DATABASE attendance_db;
USE attendance_db;

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50),
    password VARCHAR(50),
    userType VARCHAR(20)
);


CREATE TABLE students (
    id INT AUTO_INCREMENT PRIMARY KEY,
    rollNo VARCHAR(20),
    name VARCHAR(100)
);

CREATE TABLE attendance (
    id INT AUTO_INCREMENT PRIMARY KEY,
    subject VARCHAR(50),
    date DATE,
    rollNo VARCHAR(20),
    name VARCHAR(100),
    present BOOLEAN
);

CREATE TABLE reminders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    task VARCHAR(100),
    time DATETIME
);
